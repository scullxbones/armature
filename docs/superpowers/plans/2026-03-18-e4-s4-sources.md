```go
// internal/sources/types.go
package sources

import "encoding/json"

// Provider fetches source document content and reports a version signal.
type Provider interface {
	// Fetch retrieves the current content of the source as normalized markdown.
	Fetch(entry SourceEntry) (content string, sha string, versionID string, err error)
}

// SourceEntry describes a registered source document.
type SourceEntry struct {
	ID        string `json:"id"`
	Provider  string `json:"provider"` // "filesystem", "confluence", "sharepoint"
	Path      string `json:"path,omitempty"`       // filesystem: absolute path
	BaseURL   string `json:"base_url,omitempty"`   // confluence: base URL
	SpaceKey  string `json:"space_key,omitempty"`  // confluence: space key
	PageID    string `json:"page_id,omitempty"`    // confluence/sharepoint: page/item ID
	SiteURL   string `json:"site_url,omitempty"`   // sharepoint: site URL
	DriveID   string `json:"drive_id,omitempty"`   // sharepoint: drive ID
	SHA       string `json:"sha,omitempty"`        // last known SHA-256 of content
	VersionID string `json:"version_id,omitempty"` // provider-specific version token
}

// Manifest holds all registered source documents.
type Manifest struct {
	Sources []SourceEntry `json:"sources"`
}

// Marshal serializes the manifest to JSON bytes.
func (m Manifest) Marshal() ([]byte, error) {
	return json.MarshalIndent(m, "", "  ")
}

// UnmarshalManifest deserializes a manifest from JSON bytes.
func UnmarshalManifest(data []byte) (Manifest, error) {
	var m Manifest
	return m, json.Unmarshal(data, &m)
}

// Get returns the SourceEntry with the given ID, or false if not found.
func (m Manifest) Get(id string) (SourceEntry, bool) {
	for _, s := range m.Sources {
		if s.ID == id {
			return s, true
		}
	}
	return SourceEntry{}, false
}

// Upsert adds or replaces a SourceEntry by ID.
func (m *Manifest) Upsert(entry SourceEntry) {
	for i, s := range m.Sources {
		if s.ID == entry.ID {
			m.Sources[i] = entry
			return
		}
	}
	m.Sources = append(m.Sources, entry)
}
```

```go
// internal/sources/manifest.go
package sources

import (
	"encoding/json"
	"os"
	"path/filepath"
)

// manifestPath returns the path to manifest.json within issuesDir.
func manifestPath(issuesDir string) string {
	return filepath.Join(issuesDir, "state", "sources", "manifest.json")
}

// cachePath returns the directory for cached source content.
func cacheDir(issuesDir string) string {
	return filepath.Join(issuesDir, "state", "sources", "cache")
}

// CachedContentPath returns the path for the cached markdown of sourceID.
func CachedContentPath(issuesDir, sourceID string) string {
	return filepath.Join(cacheDir(issuesDir), sourceID+".md")
}

// ReadManifest loads manifest.json from .armature/state/sources/manifest.json.
// Returns empty manifest if the file does not exist.
func ReadManifest(issuesDir string) (Manifest, error) {
	path := manifestPath(issuesDir)
	data, err := os.ReadFile(path)
	if err != nil {
		if os.IsNotExist(err) {
			return Manifest{}, nil
		}
		return Manifest{}, err
	}
	return UnmarshalManifest(data)
}

// WriteManifest writes manifest.json, creating parent directories as needed.
func WriteManifest(issuesDir string, m Manifest) error {
	dir := filepath.Dir(manifestPath(issuesDir))
	if err := os.MkdirAll(dir, 0755); err != nil {
		return err
	}
	if err := os.MkdirAll(cacheDir(issuesDir), 0755); err != nil {
		return err
	}
	data, err := json.MarshalIndent(m, "", "  ")
	if err != nil {
		return err
	}
	return os.WriteFile(manifestPath(issuesDir), data, 0644)
}

// WriteCache writes content to the cache file for sourceID.
func WriteCache(issuesDir, sourceID, content string) error {
	if err := os.MkdirAll(cacheDir(issuesDir), 0755); err != nil {
		return err
	}
	return os.WriteFile(CachedContentPath(issuesDir, sourceID), []byte(content), 0644)
}

// ReadCache reads the cached markdown for sourceID.
func ReadCache(issuesDir, sourceID string) (string, error) {
	data, err := os.ReadFile(CachedContentPath(issuesDir, sourceID))
	if err != nil {
		return "", err
	}
	return string(data), nil
}
```

```go
// internal/sources/filesystem.go
package sources

import (
	"fmt"
	"os"
)

// FilesystemProvider reads source documents from the local filesystem.
type FilesystemProvider struct{}

// Fetch reads the file at entry.Path and returns its content and SHA-256.
// versionID is always empty for filesystem sources.
func (FilesystemProvider) Fetch(entry SourceEntry) (content, sha, versionID string, err error) {
	if entry.Path == "" {
		return "", "", "", fmt.Errorf("filesystem provider: entry %q has no path", entry.ID)
	}
	data, err := os.ReadFile(entry.Path)
	if err != nil {
		return "", "", "", fmt.Errorf("filesystem provider: read %s: %w", entry.Path, err)
	}
	return string(data), Fingerprint(data), "", nil
}
```

```go
// internal/sources/confluence.go
package sources

import (
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"regexp"
	"strings"
)

// ConfluenceProvider fetches pages from Confluence Cloud/Server REST API.
type ConfluenceProvider struct {
	cfg    ConfluenceConfig
	client *http.Client
}

// NewConfluenceProvider creates a Confluence provider with the given config.
func NewConfluenceProvider(cfg ConfluenceConfig) *ConfluenceProvider {
	return &ConfluenceProvider{cfg: cfg, client: &http.Client{}}
}

type confluencePage struct {
	Title   string `json:"title"`
	Version struct {
		Number int `json:"number"`
	} `json:"version"`
	Body struct {
		Storage struct {
			Value string `json:"value"`
		} `json:"storage"`
	} `json:"body"`
}

// Fetch retrieves the Confluence page at entry.PageID and returns normalized markdown.
func (p *ConfluenceProvider) Fetch(entry SourceEntry) (content, sha, versionID string, err error) {
	url := fmt.Sprintf("%s/rest/api/content/%s?expand=body.storage,version", entry.BaseURL, entry.PageID)
	req, err := http.NewRequest("GET", url, nil)
	if err != nil {
		return "", "", "", fmt.Errorf("confluence: build request: %w", err)
	}
	if p.cfg.Token != "" {
		req.Header.Set("Authorization", "Bearer "+p.cfg.Token)
	}
	req.Header.Set("Accept", "application/json")

	resp, err := p.client.Do(req)
	if err != nil {
		return "", "", "", fmt.Errorf("confluence: fetch page %s: %w", entry.PageID, err)
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		return "", "", "", fmt.Errorf("confluence: unexpected status %d for page %s", resp.StatusCode, entry.PageID)
	}

	body, err := io.ReadAll(resp.Body)
	if err != nil {
		return "", "", "", fmt.Errorf("confluence: read body: %w", err)
	}

	var page confluencePage
	if err := json.Unmarshal(body, &page); err != nil {
		return "", "", "", fmt.Errorf("confluence: parse response: %w", err)
	}

	md := htmlToMarkdown(page.Body.Storage.Value)
	if page.Title != "" {
		md = "# " + page.Title + "\n\n" + md
	}
	return md, Fingerprint([]byte(md)), fmt.Sprintf("%d", page.Version.Number), nil
}

// htmlToMarkdown strips HTML tags and returns plain text (minimal conversion).
// Full Mermaid sidecar extraction is deferred.
var tagRe = regexp.MustCompile(`<[^>]+>`)

func htmlToMarkdown(html string) string {
	s := tagRe.ReplaceAllString(html, "")
	s = strings.ReplaceAll(s, "&amp;", "&")
	s = strings.ReplaceAll(s, "&lt;", "<")
	s = strings.ReplaceAll(s, "&gt;", ">")
	s = strings.ReplaceAll(s, "&nbsp;", " ")
	s = strings.TrimSpace(s)
	return s
}
```

```go
// internal/sources/sharepoint.go
package sources

import (
	"fmt"
	"io"
	"net/http"
	"strings"
)

// SharePointProvider fetches files from SharePoint via Graph API.
type SharePointProvider struct {
	cfg    SharePointConfig
	client *http.Client
}

// NewSharePointProvider creates a SharePoint provider with the given config.
func NewSharePointProvider(cfg SharePointConfig) *SharePointProvider {
	return &SharePointProvider{cfg: cfg, client: &http.Client{}}
}

// Fetch retrieves the SharePoint file at entry.PageID (drive item path) and returns its content.
// versionID is set from the ETag response header.
func (p *SharePointProvider) Fetch(entry SourceEntry) (content, sha, versionID string, err error) {
	// Graph API: GET /sites/{site-id}/drives/{drive-id}/root:{item-path}:/content
	url := entry.SiteURL + entry.PageID
	req, err := http.NewRequest("GET", url, nil)
	if err != nil {
		return "", "", "", fmt.Errorf("sharepoint: build request: %w", err)
	}
	if p.cfg.Token != "" {
		req.Header.Set("Authorization", "Bearer "+p.cfg.Token)
	}

	resp, err := p.client.Do(req)
	if err != nil {
		return "", "", "", fmt.Errorf("sharepoint: fetch item %s: %w", entry.PageID, err)
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		return "", "", "", fmt.Errorf("sharepoint: unexpected status %d for item %s", resp.StatusCode, entry.PageID)
	}

	body, err := io.ReadAll(resp.Body)
	if err != nil {
		return "", "", "", fmt.Errorf("sharepoint: read body: %w", err)
	}

	etag := strings.Trim(resp.Header.Get("ETag"), `"`)
	md := string(body)
	return md, Fingerprint(body), etag, nil
}
```

```go
		pageID   string
		siteURL  string
		driveID  string
	)
	cmd := &cobra.Command{
		Use:   "add <source-id>",
		Short: "Register a source document",
		Args:  cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			sourceID := args[0]
			issuesDir := appCtx.IssuesDir

			entry := sources.SourceEntry{
				ID:       sourceID,
				Provider: provider,
				Path:     path,
				BaseURL:  baseURL,
				SpaceKey: spaceKey,
				PageID:   pageID,
				SiteURL:  siteURL,
				DriveID:  driveID,
			}

			p, err := providerFor(entry)
			if err != nil {
				return err
			}

			content, sha, versionID, err := p.Fetch(entry)
			if err != nil {
				return fmt.Errorf("fetch source: %w", err)
			}
			entry.SHA = sha
			entry.VersionID = versionID

			if err := sources.WriteCache(issuesDir, sourceID, content); err != nil {
				return fmt.Errorf("write cache: %w", err)
			}

			manifest, err := sources.ReadManifest(issuesDir)
			if err != nil {
				return fmt.Errorf("read manifest: %w", err)
			}
			manifest.Upsert(entry)
			if err := sources.WriteManifest(issuesDir, manifest); err != nil {
				return fmt.Errorf("write manifest: %w", err)
			}

			workerID, logPath, err := resolveWorkerAndLog()
			if err != nil {
				return fmt.Errorf("worker not initialized: %w", err)
			}
			if err := emitSourceFingerprintOp(workerID, logPath, sourceID, sha, versionID, provider); err != nil {
				return fmt.Errorf("emit fingerprint op: %w", err)
			}

			_, _ = fmt.Fprintf(cmd.OutOrStdout(), "added source %q (%s) sha=%s\n", sourceID, provider, sha[:8])
			return nil
		},
	}
	cmd.Flags().StringVar(&provider, "provider", "filesystem", "Provider: filesystem|confluence|sharepoint")
	cmd.Flags().StringVar(&path, "path", "", "Filesystem path (filesystem provider)")
	cmd.Flags().StringVar(&baseURL, "base-url", "", "Base URL (confluence provider)")
	cmd.Flags().StringVar(&spaceKey, "space-key", "", "Space key (confluence provider)")
	cmd.Flags().StringVar(&pageID, "page-id", "", "Page/item ID (confluence/sharepoint)")
	cmd.Flags().StringVar(&siteURL, "site-url", "", "Site URL (sharepoint provider)")
	cmd.Flags().StringVar(&driveID, "drive-id", "", "Drive ID (sharepoint provider)")
	return cmd
}

func newSourcesSyncCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "sync",
		Short: "Sync all registered sources and update cache",
		RunE: func(cmd *cobra.Command, args []string) error {
			issuesDir := appCtx.IssuesDir
			workerID, logPath, err := resolveWorkerAndLog()
			if err != nil {
				return err
			}

			manifest, err := sources.ReadManifest(issuesDir)
			if err != nil {
				return fmt.Errorf("read manifest: %w", err)
			}

			changed := 0
			for _, entry := range manifest.Sources {
				p, err := providerFor(entry)
				if err != nil {
					_, _ = fmt.Fprintf(cmd.ErrOrStderr(), "warning: skipping %s: %v\n", entry.ID, err)
					continue
				}
				content, sha, versionID, err := p.Fetch(entry)
				if err != nil {
					_, _ = fmt.Fprintf(cmd.ErrOrStderr(), "warning: fetch %s: %v\n", entry.ID, err)
					continue
				}
				if sha == entry.SHA {
					_, _ = fmt.Fprintf(cmd.OutOrStdout(), "  unchanged: %s\n", entry.ID)
					continue
				}
				entry.SHA = sha
				entry.VersionID = versionID
				manifest.Upsert(entry)
				if err := sources.WriteCache(issuesDir, entry.ID, content); err != nil {
					return fmt.Errorf("write cache %s: %w", entry.ID, err)
				}
				if err := emitSourceFingerprintOp(workerID, logPath, entry.ID, sha, versionID, entry.Provider); err != nil {
					return fmt.Errorf("emit fingerprint op %s: %w", entry.ID, err)
				}
				_, _ = fmt.Fprintf(cmd.OutOrStdout(), "  updated:   %s sha=%s\n", entry.ID, sha[:8])
				changed++
			}

			if err := sources.WriteManifest(issuesDir, manifest); err != nil {
				return fmt.Errorf("write manifest: %w", err)
			}
			_, _ = fmt.Fprintf(cmd.OutOrStdout(), "synced %d source(s), %d changed\n", len(manifest.Sources), changed)
			return nil
		},
	}
}

func newSourcesVerifyCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "verify",
		Short: "Verify cached sources match manifest SHAs",
		RunE: func(cmd *cobra.Command, args []string) error {
			issuesDir := appCtx.IssuesDir
			manifest, err := sources.ReadManifest(issuesDir)
			if err != nil {
				return fmt.Errorf("read manifest: %w", err)
			}

			ok := true
			for _, entry := range manifest.Sources {
				cached, err := sources.ReadCache(issuesDir, entry.ID)
				if err != nil {
					_, _ = fmt.Fprintf(cmd.OutOrStdout(), "MISSING  %s\n", entry.ID)
					ok = false
					continue
				}
				actual := sources.Fingerprint([]byte(cached))
				if actual != entry.SHA {
					_, _ = fmt.Fprintf(cmd.OutOrStdout(), "MISMATCH %s (manifest=%s cache=%s)\n",
						entry.ID, entry.SHA[:8], actual[:8])
					ok = false
				} else {
					_, _ = fmt.Fprintf(cmd.OutOrStdout(), "OK       %s\n", entry.ID)
				}
			}
			if !ok {
				return fmt.Errorf("source verification failed")
			}
			return nil
		},
	}
}

// providerFor returns the appropriate Provider for the given SourceEntry.
func providerFor(entry sources.SourceEntry) (sources.Provider, error) {
	switch entry.Provider {
	case "filesystem", "":
		return sources.FilesystemProvider{}, nil
	case "confluence":
		cfg := sources.ConfluenceConfigFromEnv()
		return sources.NewConfluenceProvider(cfg), nil
	case "sharepoint":
		cfg := sources.SharePointConfigFromEnv()
		return sources.NewSharePointProvider(cfg), nil
	default:
		return nil, fmt.Errorf("unknown provider %q", entry.Provider)
	}
}

// emitSourceFingerprintOp writes a source-fingerprint op to the ops log.
func emitSourceFingerprintOp(workerID, logPath, sourceID, sha, versionID, provider string) error {
	o := ops.Op{
		Type:      ops.OpSourceFingerprint,
		TargetID:  sourceID,
		Timestamp: nowEpoch(),
		WorkerID:  workerID,
		Payload: ops.Payload{
			SHA:       sha,
			VersionID: versionID,
			Provider:  provider,
		},
	}
	return appendLowStakesOp(logPath, o)
}
```
